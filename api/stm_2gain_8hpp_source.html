26  * @brief GainSTM provides a function to display Gain sequentially and periodically.
27  * @details GainSTM uses a timer on the FPGA to ensure that Gain is precisely timed.
28  * GainSTM currently has the following three limitations.

30  */

39  void add(G& gain) {
40  if (_gains.size() + 1 > driver::GAIN_STM_BUF_SIZE_MAX) throw std::runtime_error("GainSTM out of buffer");
41 
42  gain.build(_geometry);
43 
44  _gains.emplace_back(gain.drives());
45  }
