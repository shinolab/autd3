66  gain_stm_legacy_header(tx);
67 
68  if (is_finished()) return;
69 
70  const auto is_first_frame = _sent == 0;
71 
72  if (is_first_frame) {
73  gain_stm_legacy_body({}, is_first_frame, _freq_div, false, _mode, tx);
74  _sent += 1;
75  return;
76  }
77 
78  bool is_last_frame;
79  driver::LegacyDrive *p1, *p2, *p3, *p4;
80  switch (_mode) {
81  case driver::Mode::PhaseDutyFull:
82  is_last_frame = _sent + 1 == _gains.size() + 1;
83  gain_stm_legacy_body({_gains.at(_sent - 1).data.data()}, is_first_frame, _freq_div, is_last_frame, _mode, tx);
84  _sent += 1;
85  break;
86  case driver::Mode::PhaseFull:
87  is_last_frame = _sent + 2 >= _gains.size() + 1;
88  p1 = _gains.at(_sent - 1).data.data();
89  p2 = _sent + 1 - 1 < _gains.size() ? _gains.at(_sent + 1 - 1).data.data() : nullptr;
90  gain_stm_legacy_body({p1, p2}, is_first_frame, _freq_div, is_last_frame, _mode, tx);
91  _sent += 2;
92  break;
93  case driver::Mode::PhaseHalf:
94  is_last_frame = _sent + 4 >= _gains.size() + 1;
95  p1 = _gains.at(_sent - 1).data.data();
96  p2 = _sent + 1 - 1 < _gains.size() ? _gains.at(_sent + 1 - 1).data.data() : nullptr;
97  p3 = _sent + 2 - 1 < _gains.size() ? _gains.at(_sent + 2 - 1).data.data() : nullptr;
98  p4 = _sent + 3 - 1 < _gains.size() ? _gains.at(_sent + 3 - 1).data.data() : nullptr;
99  gain_stm_legacy_body({p1, p2, p3, p4}, is_first_frame, _freq_div, is_last_frame, _mode, tx);
100  _sent += 4;
101  break;
102  }
103  }
