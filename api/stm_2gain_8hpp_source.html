24 /**
25  * @brief GainSTM provides a function to display Gain sequentially and periodically.
26  * @details GainSTM uses a timer on the FPGA to ensure that Gain is precisely timed.
27  * GainSTM currently has the following three limitations.
28  * 1. The maximum number of gains is driver::GAIN_STM_BUF_SIZE_MAX.
29  */

31 struct GainSTM {};
