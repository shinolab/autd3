71  * \return if this function returns true and check_trials > 0, it guarantees that the devices have processed the data.

73  bool synchronize() {
74  driver::force_fan(_tx_buf, force_fan);
75  driver::reads_fpga_info(_tx_buf, reads_fpga_info);

77  const auto msg_id = get_id();
78  std::vector<uint16_t> cycles;
79  std::for_each(_geometry.begin(), _geometry.end(), [&](const auto& dev) {

142  * @brief Enumerate firmware information
143  * \return vector of driver::FirmwareInfo. If failed, the vector is empty.

145  [[nodiscard]] std::vector<driver::FirmwareInfo> firmware_infos() {
146  std::vector<driver::FirmwareInfo> firmware_infos;
147 
148  const auto pack_ack = [&]() -> std::vector<uint8_t> {
149  std::vector<uint8_t> acks;
150  if (!_link->send(_tx_buf)) return acks;
151  if (wait_msg_processed(200) == 200) return acks;
152  std::transform(_rx_buf.begin(), _rx_buf.end(), std::back_inserter(acks), [](driver::RxMessage msg) noexcept { return msg.ack; });
153  return acks;
154  };
155 
156  cpu_version(_tx_buf);
157  const auto cpu_versions = pack_ack();
158  if (cpu_versions.empty()) return firmware_infos;
159 
160  fpga_version(_tx_buf);
161  const auto fpga_versions = pack_ack();
162  if (fpga_versions.empty()) return firmware_infos;
163 
164  fpga_functions(_tx_buf);
165  const auto fpga_functions = pack_ack();
166  if (fpga_functions.empty()) return firmware_infos;
167 
168  for (size_t i = 0; i < _geometry.num_devices(); i++)
169  firmware_infos.emplace_back(i, cpu_versions.at(i), fpga_versions.at(i), fpga_functions.at(i));
170 
171  return firmware_infos;

212  /**
213  * @brief Send header and body data to devices
214  * \return if this function returns true and check_trials > 0, it guarantees that the devices have processed the data.
215  */
216  template <typename H, typename B>
217  auto send(H& header, B& body) ->
218  typename std::enable_if_t<std::is_base_of_v<core::DatagramHeader, H> && std::is_base_of_v<core::DatagramBody<T>, B>, bool> {
219  header.init();
220  body.init();
221 
222  driver::force_fan(_tx_buf, force_fan);
223  driver::reads_fpga_info(_tx_buf, reads_fpga_info);
224 
225  while (true) {
226  const auto msg_id = get_id();
227  header.pack(msg_id, _tx_buf);
228  body.pack(_geometry, _tx_buf);
229  _link->send(_tx_buf);
230  const auto trials = wait_msg_processed(check_trials);
231  if ((check_trials != 0) && (trials == check_trials)) return false;
232  if (header.is_finished() && body.is_finished()) break;
233  if (trials == 0) std::this_thread::sleep_for(std::chrono::microseconds(send_interval * driver::EC_SYNC0_CYCLE_TIME_MICRO_SEC));
234  }
235  return true;
236  }

270  static uint8_t get_id() noexcept {
271  static std::atomic id_body{driver::MSG_BEGIN};
272  if (uint8_t expected = driver::MSG_END; !id_body.compare_exchange_weak(expected, driver::MSG_BEGIN)) id_body.fetch_add(0x01);
273  return id_body.load();
274  }

276  size_t wait_msg_processed(const size_t max_trial) {
277  size_t i;
278  const auto msg_id = _tx_buf.header().msg_id;
279  for (i = 0; i < max_trial; i++) {
280  if (_link->receive(_rx_buf) && _rx_buf.is_msg_processed(msg_id)) break;
281  std::this_thread::sleep_for(std::chrono::microseconds(send_interval * driver::EC_SYNC0_CYCLE_TIME_MICRO_SEC));
282  }
283  return i;
284  }
