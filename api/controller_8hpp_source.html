245  static std::atomic id_body{driver::MSG_BEGIN};
246  if (uint8_t expected = driver::MSG_END; !id_body.compare_exchange_weak(expected, driver::MSG_BEGIN)) id_body.fetch_add(0x01);

250  bool wait_msg_processed(const size_t max_trial) {
251  if (!check_ack) return true;
252 
253  const auto msg_id = _tx_buf.header().msg_id;
254  const auto wait = static_cast<uint64_t>(std::ceil(driver::EC_TRAFFIC_DELAY * 1000.0 / static_cast<double>(driver::EC_DEVICE_PER_FRAME) *
255  static_cast<double>(_geometry.num_devices())));
256  for (size_t i = 0; i < max_trial; i++) {
257  if (!_link->receive(_rx_buf)) continue;
258  if (_rx_buf.is_msg_processed(msg_id)) return true;
259  std::this_thread::sleep_for(std::chrono::milliseconds(wait));
260  }
261  return false;
262  }
