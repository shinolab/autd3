21 struct Device {
22  explicit Device(const size_t id, const Vector3& position, const Quaternion& rotation) : _origin(position) {
23  const Eigen::Transform<double, 3, Eigen::Affine> transform_matrix = Eigen::Translation<double, 3>(position) * rotation;
24  const Vector3 x_direction = rotation * Vector3(1, 0, 0);
25  const Vector3 y_direction = rotation * Vector3(0, 1, 0);
26  const Vector3 z_direction = rotation * Vector3(0, 0, 1);
27 
28  _transducers.reserve(driver::NUM_TRANS_IN_UNIT);
29  size_t i = id * driver::NUM_TRANS_IN_UNIT;
30  for (size_t y = 0; y < driver::NUM_TRANS_Y; y++)
31  for (size_t x = 0; x < driver::NUM_TRANS_X; x++) {
32  if (driver::is_missing_transducer(x, y)) continue;
33  const auto local_pos =
34  Vector4(static_cast<double>(x) * driver::TRANS_SPACING_MM, static_cast<double>(y) * driver::TRANS_SPACING_MM, 0.0, 1.0);
35  const Vector4 global_pos = transform_matrix * local_pos;
36  _transducers.emplace_back(i++, global_pos.head<3>(), x_direction, y_direction, z_direction);
37  }
38  _trans_inv = transform_matrix.inverse();
39  }
