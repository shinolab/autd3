31  explicit Device(const size_t id, const Vector3& position, const Quaternion& rotation) : _origin(position) {
32  const Eigen::Transform<double, 3, Eigen::Affine> transform_matrix = Eigen::Translation<double, 3>(position) * rotation;
33  const Vector3 x_direction = rotation * Vector3(1, 0, 0);
34  const Vector3 y_direction = rotation * Vector3(0, 1, 0);
35  const Vector3 z_direction = rotation * Vector3(0, 0, 1);
36 
37  _transducers.reserve(driver::NUM_TRANS_IN_UNIT);
38  size_t i = id * driver::NUM_TRANS_IN_UNIT;
39  for (size_t y = 0; y < driver::NUM_TRANS_Y; y++)
40  for (size_t x = 0; x < driver::NUM_TRANS_X; x++) {
41  if (driver::is_missing_transducer(x, y)) continue;
42  const auto local_pos =
43  Vector4(static_cast<double>(x) * driver::TRANS_SPACING_MM, static_cast<double>(y) * driver::TRANS_SPACING_MM, 0.0, 1.0);
44  const Vector4 global_pos = transform_matrix * local_pos;
45  _transducers.emplace_back(i++, global_pos.head<3>(), x_direction, y_direction, z_direction);
46  }
47  _trans_inv = transform_matrix.inverse();
48  }
