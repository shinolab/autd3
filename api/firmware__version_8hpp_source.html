72  [[nodiscard]] static std::string firmware_version_map(const uint8_t version_num) {
73  std::stringstream ss;
74  if (version_num == 0) return "older than v0.4";
75  if (version_num <= 0x06) {
76  ss << "v0." << version_num + 3;
77  return ss.str();
78  }
79  if (version_num <= 0x09) return "unknown";
80  if (version_num <= 0x15) {
81  ss << "v1." << version_num - 0x0A;
82  return ss.str();
83  }
84  if (version_num <= 0x82) {
85  ss << "v2." << version_num - 0x80;
86  return ss.str();
87  }
88  if (version_num == 0xFF) return "emulator";
89 
90  ss << "unknown (" << std::hex << static_cast<int>(version_num) << ")";
91  return ss.str();
92  }
