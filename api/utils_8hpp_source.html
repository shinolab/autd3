36  /**
37  * \brief Directivity of T4010A1
38  * \param theta_deg zenith angle in degree
39  * \return directivity
40  */
41  static double t4010a1(double theta_deg) {
42  theta_deg = std::abs(theta_deg);
43  while (theta_deg > 90) theta_deg = std::abs(180 - theta_deg);
44  const auto i = static_cast<size_t>(std::ceil(theta_deg / 10));
45  if (i == 0) return 1;
46  const auto a = DIR_COEFFICIENT_A[i - 1];
47  const auto b = DIR_COEFFICIENT_B[i - 1];
48  const auto c = DIR_COEFFICIENT_C[i - 1];
49  const auto d = DIR_COEFFICIENT_D[i - 1];
50  const auto x = theta_deg - static_cast<double>(i - 1) * 10;
51  return a + b * x + c * x * x + d * x * x * x;
52  }
53 };

62 inline std::complex<double> propagate(const core::Vector3& source_pos, const core::Vector3& source_dir, const double attenuation,
63  const double wavenumber, const core::Vector3& target) {
64  const auto diff = target - source_pos;
65  const auto dist = diff.norm();
66 
67  const auto theta = std::atan2(source_dir.cross(diff).norm(), source_dir.dot(diff)) * 180.0 / driver::pi;
68 
69  const auto d = Directivity::t4010a1(theta);
70  const auto r = d * std::exp(-dist * attenuation) / dist;
71  const auto phi = -wavenumber * dist;
72  return r * std::exp(std::complex<double>(0., phi));
73 }
