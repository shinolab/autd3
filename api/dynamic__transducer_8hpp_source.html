32 struct DynamicDriveData final : DriveData<T> {
33  void init(const size_t size) override {
34  legacy_drives.resize(size, driver::LegacyDrive{0x00, 0x00});
35  duties.resize(size, driver::Duty{0x0000});
36  phases.resize(size, driver::Phase{0x0000});
37  }
38  void set_drive(const T& tr, const double phase, const double amp) override {
39  switch (T::mode()) {
40  case TransducerMode::Legacy:
41  legacy_drives.at(tr.id()).set(amp, phase);
42  break;
43  case TransducerMode::Normal:
44  duties.at(tr.id()).set(amp, tr.cycle());
45  phases.at(tr.id()).set(phase, tr.cycle());
46  break;
47  case TransducerMode::NormalPhase:
48  phases.at(tr.id()).set(phase, tr.cycle());
49  break;
50  default:
51  legacy_drives.at(tr.id()).set(amp, phase);
52  break;
53  }
54  }

56  void copy_from(size_t idx, const typename T::D& src) override {
57  const auto* s = src.legacy_drives.data() + idx * driver::NUM_TRANS_IN_UNIT;
58  auto* d = legacy_drives.data() + idx * driver::NUM_TRANS_IN_UNIT;
59  std::memcpy(d, s, sizeof(driver::LegacyDrive) * driver::NUM_TRANS_IN_UNIT);
60 
61  auto ds = src.duties.data() + idx * driver::NUM_TRANS_IN_UNIT;
62  const auto dd = duties.data() + idx * driver::NUM_TRANS_IN_UNIT;
63  std::memcpy(dd, ds, sizeof(driver::Duty) * driver::NUM_TRANS_IN_UNIT);
64  auto ps = src.phases.data() + idx * driver::NUM_TRANS_IN_UNIT;
65  const auto pd = phases.data() + idx * driver::NUM_TRANS_IN_UNIT;
66  std::memcpy(pd, ps, sizeof(driver::Phase) * driver::NUM_TRANS_IN_UNIT);
67  }

106  static void pack_body(bool& phase_sent, bool& duty_sent, const D& drives, driver::TxDatagram& tx) noexcept {
107  switch (_mode) {
108  case TransducerMode::Legacy:
109  normal_legacy_body(drives.legacy_drives.data(), tx);
110  phase_sent = true;
111  duty_sent = true;
112  break;
113 
114  case TransducerMode::Normal:
115  if (!phase_sent) {
116  normal_phase_body(drives.phases.data(), tx);
117  phase_sent = true;
118  } else {
119  normal_duty_body(drives.duties.data(), tx);
120  duty_sent = true;
121  }
122  break;
123  case TransducerMode::NormalPhase:
124  normal_phase_body(drives.phases.data(), tx);
125  phase_sent = true;
126  duty_sent = true;
127  break;
128  default:
129  normal_legacy_body(drives.legacy_drives.data(), tx);
130  phase_sent = true;
131  duty_sent = true;
132  break;
133  }
