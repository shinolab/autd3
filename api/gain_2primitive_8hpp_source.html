77 template <typename T = core::LegacyTransducer, std::enable_if_t<std::is_base_of_v<core::Transducer<typename T::D>, T>, nullptr_t> = nullptr>
78 class BesselBeam final : public core::Gain<T> {
79  public:
80  /**
81  * @param[in] apex apex of the conical wavefront of the beam
82  * @param[in] vec_n direction of the beam
83  * @param[in] theta_z angle between the side of the cone and the plane perpendicular to direction of the beam
84  * @param[in] amp amplitude of the wave (from 0.0 to 1.0)
85  */
86  explicit BesselBeam(core::Vector3 apex, core::Vector3 vec_n, const double theta_z, const double amp = 1.0)
87  : core::Gain<T>(), _apex(std::move(apex)), _vec_n(std::move(vec_n)), _theta_z(theta_z), _amp(amp) {}
88 
89  void calc(const core::Geometry<T>& geometry) override {
90  _vec_n.normalize();
91  core::Vector3 v = core::Vector3::UnitZ().cross(_vec_n);
92  const auto theta_v = std::asin(v.norm());
93  v.normalize();
94  const Eigen::AngleAxisd rot(-theta_v, v);
95 
96  std::for_each(geometry.begin(), geometry.end(), [&](const auto& dev) {
97  std::for_each(dev.begin(), dev.end(), [&](const auto& transducer) {
98  const auto r = transducer.position() - this->_apex;
99  const auto rr = rot * r;
100  const auto d = std::sin(_theta_z) * std::sqrt(rr.x() * rr.x() + rr.y() * rr.y()) - std::cos(_theta_z) * rr.z();
101  const auto phase = transducer.align_phase_at(d, geometry.sound_speed);
102  this->_props.drives.set_drive(transducer, phase, _amp);
103  });
104  });
105  }
