80 class BesselBeam final : public core::Gain {
81  public:
82  /**
83  * @param[in] apex apex of the conical wavefront of the beam
84  * @param[in] vec_n direction of the beam
85  * @param[in] theta_z angle between the side of the cone and the plane perpendicular to direction of the beam
86  * @param[in] amp amplitude of the wave (from 0.0 to 1.0)
87  */
88  explicit BesselBeam(core::Vector3 apex, core::Vector3 vec_n, const double theta_z, const double amp = 1.0)
89  : core::Gain(), _apex(std::move(apex)), _vec_n(std::move(vec_n)), _theta_z(theta_z), _amp(amp) {}
90 
91  void calc(const core::Geometry& geometry) override {
92  _vec_n.normalize();
93  core::Vector3 v = core::Vector3::UnitZ().cross(_vec_n);
94  const auto theta_v = std::asin(v.norm());
95  v.normalize();
96  const Eigen::AngleAxisd rot(-theta_v, v);
97 
98  std::for_each(geometry.begin(), geometry.end(), [&](const auto& dev) {
99  std::for_each(dev.begin(), dev.end(), [&](const auto& transducer) {
100  const auto r = transducer.position() - this->_apex;
101  const auto rr = rot * r;
102  const auto d = std::sin(_theta_z) * std::sqrt(rr.x() * rr.x() + rr.y() * rr.y()) - std::cos(_theta_z) * rr.z();
103  const auto phase = transducer.align_phase_at(d, geometry.sound_speed);
104  _drives[transducer.id()].amp = _amp;
105  _drives[transducer.id()].phase = phase;
106  });
107  });
108  }
109

114  BesselBeam& operator=(BesselBeam&& obj) = default;
