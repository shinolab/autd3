28 template <typename T>
29 struct NormalDriveData final : DriveData<T> {
30  void init(const size_t size) override {
31  duties.resize(size, driver::Duty{0x0000});
32  phases.resize(size, driver::Phase{0x0000});
33  }
34  void set_drive(const T& tr, const double phase, const double amp) override {
35  duties.at(tr.id()).duty = static_cast<uint16_t>(static_cast<double>(tr.cycle()) * std::asin(amp) / driver::pi);
36  phases.at(tr.id()).phase = static_cast<uint16_t>(
37  rem_euclid(static_cast<int32_t>(std::round(phase * static_cast<double>(tr.cycle()))), static_cast<int32_t>(tr.cycle())));
38  }
39  void copy_from(size_t idx, const typename T::D& src) override {
40  auto ds = src.duties.data() + idx * driver::NUM_TRANS_IN_UNIT;
41  const auto dd = duties.data() + idx * driver::NUM_TRANS_IN_UNIT;
42  std::memcpy(dd, ds, sizeof(driver::Duty) * driver::NUM_TRANS_IN_UNIT);
43  auto ps = src.phases.data() + idx * driver::NUM_TRANS_IN_UNIT;
44  const auto pd = phases.data() + idx * driver::NUM_TRANS_IN_UNIT;
45  std::memcpy(pd, ps, sizeof(driver::Phase) * driver::NUM_TRANS_IN_UNIT);
46  }

48  std::vector<driver::Duty> duties{};
49  std::vector<driver::Phase> phases{};
50 };
