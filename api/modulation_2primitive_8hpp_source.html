54  * @param[in] amp peek to peek ultrasound amplitude (Maximum value is 1.0)
55  * @param[in] offset offset of ultrasound amplitude
56  * @details Ultrasound amplitude oscillate from offset-amp/2 to offset+amp/2.
57  * If the value exceeds the range of [0, 1], the value will be clamped in the [0, 1].

59  explicit Sine(const int freq, const double amp = 1.0, const double offset = 0.5) noexcept : Modulation(), _freq(freq), _amp(amp), _offset(offset) {}
60 
61  void calc() override {
62  const auto f_s = static_cast<int32_t>(sampling_frequency());
63 
64  const auto f = std::clamp(this->_freq, 1, f_s / 2);
65 
66  const auto k = std::gcd(f_s, f);
67 
68  const size_t n = f_s / k;
69  const size_t d = f / k;
70 
71  this->_props.buffer.resize(n, 0);
72  for (size_t i = 0; i < n; i++) {
73  const auto amp = this->_amp / 2.0 * std::sin(2.0 * driver::pi * static_cast<double>(d * i) / static_cast<double>(n)) + this->_offset;

152  const auto f_s = sampling_frequency();
153  const auto f = std::clamp(this->_freq, f_s / static_cast<double>(driver::MOD_BUF_SIZE_MAX), f_s / 2.0);
154

182 
183  void calc() override {
184  const auto f_s = static_cast<int32_t>(sampling_frequency());
185  const auto f = std::clamp(this->_freq, 1, f_s / 2);
186  const auto k = std::gcd(f_s, f);
187  const size_t n = f_s / k;
188  const size_t d = f / k;
189 
190  const auto low = static_cast<uint8_t>(std::round(std::asin(std::clamp(_low, 0.0, 1.0)) / driver::pi * 510.0));
191  std::fill(this->_props.buffer.begin(), this->_props.buffer.end(), low);
192  this->_props.buffer.resize(n, low);
193 
194  const auto high = static_cast<uint8_t>(std::round(std::asin(std::clamp(_high, 0.0, 1.0)) / driver::pi * 510.0));
195  auto* cursor = this->_props.buffer.data();
196  for (size_t i = 0; i < d; i++) {
197  const size_t size = (n + i) / d;
198  std::memset(cursor, high, static_cast<size_t>(std::round(static_cast<double>(size) * _duty)));
199  cursor += size;
200  }
201  }
202 
203  private:
204  int _freq;
205  double _low;
206  double _high;
207  double _duty;
208 };
