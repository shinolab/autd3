216 inline void gain_stm_legacy_body(const std::vector<const LegacyDrive*>& drives, const bool is_first_frame, const uint32_t freq_div,
217  const bool is_last_frame, const Mode mode, TxDatagram& tx) noexcept(false) {
218  if (is_first_frame) {
219  if (freq_div < STM_SAMPLING_FREQ_DIV_MIN) {
220  std::stringstream ss;
221  ss << "STM frequency division is oud of range. Minimum is " << STM_SAMPLING_FREQ_DIV_MIN << ", but you use " << freq_div;
222  throw std::runtime_error(ss.str());
223  }
224 
225  tx.header().cpu_flag.set(CPUControlFlags::STM_BEGIN);
226  for (size_t i = 0; i < tx.size(); i++) {
227  tx.bodies()[i].gain_stm_head().set_freq_div(freq_div);
228  tx.bodies()[i].gain_stm_head().set_mode(mode);
229  }
230  } else {
231  switch (mode) {
232  case Mode::PhaseDutyFull:
233  std::memcpy(reinterpret_cast<LegacyDrive*>(tx.bodies()), drives[0], sizeof(Body) * tx.size());
234  break;
235  case Mode::PhaseFull:
236  for (size_t i = 0; i < tx.size(); i++) {
237  auto* b = reinterpret_cast<PhaseFull*>(tx.bodies() + i);
238  auto* s = drives[0] + i * NUM_TRANS_IN_UNIT;
239  for (size_t j = 0; j < NUM_TRANS_IN_UNIT; j++) b[j].phase_0 = s[j].phase;
240  }
241  if (drives[1] != nullptr)
242  for (size_t i = 0; i < tx.size(); i++) {
243  auto* b = reinterpret_cast<PhaseFull*>(tx.bodies() + i);
244  auto* s = drives[1] + i * NUM_TRANS_IN_UNIT;
245  for (size_t j = 0; j < NUM_TRANS_IN_UNIT; j++) b[j].phase_1 = s[j].phase;
246  }
247  break;
248  case Mode::PhaseHalf:
249  for (size_t i = 0; i < tx.size(); i++) {
250  auto* b = reinterpret_cast<PhaseHalf*>(tx.bodies() + i);
251  auto* s = drives[0] + i * NUM_TRANS_IN_UNIT;
252  for (size_t j = 0; j < NUM_TRANS_IN_UNIT; j++) b[j].set(0, s[j].phase);
253  }
254  if (drives[1] != nullptr)
255  for (size_t i = 0; i < tx.size(); i++) {
256  auto* b = reinterpret_cast<PhaseHalf*>(tx.bodies() + i);
257  auto* s = drives[1] + i * NUM_TRANS_IN_UNIT;
258  for (size_t j = 0; j < NUM_TRANS_IN_UNIT; j++) b[j].set(1, s[j].phase);
259  }
260  if (drives[2] != nullptr)
261  for (size_t i = 0; i < tx.size(); i++) {
262  auto* b = reinterpret_cast<PhaseHalf*>(tx.bodies() + i);
263  auto* s = drives[2] + i * NUM_TRANS_IN_UNIT;
264  for (size_t j = 0; j < NUM_TRANS_IN_UNIT; j++) b[j].set(2, s[j].phase);
265  }
266  if (drives[3] != nullptr)
267  for (size_t i = 0; i < tx.size(); i++) {
268  auto* b = reinterpret_cast<PhaseHalf*>(tx.bodies() + i);
269  auto* s = drives[3] + i * NUM_TRANS_IN_UNIT;
270  for (size_t j = 0; j < NUM_TRANS_IN_UNIT; j++) b[j].set(3, s[j].phase);
271  }
272  break;
273  default:
274  std::memcpy(reinterpret_cast<LegacyDrive*>(tx.bodies()), drives[0], sizeof(Body) * tx.size());
275  break;
276  }
277  }
278 
279  tx.header().cpu_flag.set(CPUControlFlags::WRITE_BODY);
280 
281  if (is_last_frame) tx.header().cpu_flag.set(CPUControlFlags::STM_END);
282 
283  tx.num_bodies = tx.size();
